/** 
 * Core of jelly-chart 
 * @class Core
 * */
class Core {
  constructor() {
    this.setAttrs(_attrs);
    this.__execs__ = {
      autoResize: false,
      axis: {}, //axis settings
      condition: null, 
      canvas: null, 
      hidden: null,
      field: {}, // {x,y,color,raidus}
      legend:null, 
      mark: local(),
      scale: {},
      tooltip: null,
      regions: null,
      nodes: null
    }; 
    this.on = undefined; 
    this.__execs__.selectDispatch = dispatch('selectClick', 'selectEnter', 'selectLeave', 'legendEnter', 'legendLeave');
    this.rebindOnMethod(this.__execs__.selectDispatch);
  }
  
}
